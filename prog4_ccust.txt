public class Customer {
	
	String cid,username,pass;
	long phone;
	
	public Customer(String username,String pass,String cid,long phone) {
		this.cid = cid;
		this.phone = phone;
		this.pass = pass;
		this.username = username;
	}
	
	public String toString() {
		return "Username :"+username+"  CID: "+cid+"   PHONE: "+phone;
	}
}
